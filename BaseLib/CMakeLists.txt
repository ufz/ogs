# Source files

GET_SOURCE_FILES(SOURCES)

	list(REMOVE_ITEM SOURCES PETScWallClockTimer.h)
ENDIF()

# Create the library

INCLUDE_DIRECTORIES(
	.
)

TARGET_LINK_LIBRARIES( BaseLib
	${Boost_LIBRARIES}
)

IF(WIN32)
	TARGET_LINK_LIBRARIES( BaseLib WinMM )
ENDIF()
